fn parse_run(run: &str, line_no: usize) -> io::Result<u64> {
    let mut value: u128 = 0;
    for (i, pair) in run.split(':').enumerate() {
        debug_assert_eq!(pair.len(), 2);

    let bytes = line.as_bytes();

        return None;

    for start in 0..=bytes.len() - RUN_LEN {

**Part B — rewrite `parse_run` (decode.rs:94-119):**

```rust
/// Decode a validated 11-pair run into its `u64` value.
///
/// The array-typed parameter guarantees `RUN_LEN` bytes; the function
/// also validates that every non-colon position is an ASCII digit. The
/// `u128` accumulator detects overflow cleanly before the final
/// `u64::try_from`.
///
/// # Errors
///
/// Returns `io::ErrorKind::InvalidData` with one of three messages:
/// * `"line {N}: non-digit byte at pair {P}"` — byte is not `0..=9`.
/// * `"line {N}: invalid base-60 digit {D} at pair {P}"` — digit ≥ 60.
/// * `"line {N}: decoded value exceeds u64::MAX"` — overflow on final
///   `u128 → u64` conversion.
fn parse_run(run: &[u8; RUN_LEN], line_no: usize) -> io::Result<u64> {
    let mut value: u128 = 0;
    for i in 0..DIGITS {
        let pair_start = i * (PAIR + 1); // 3 bytes per pair: 2 digits + 1 colon
        let hi_byte = run[pair_start];
        let lo_byte = run[pair_start + 1];
        if !hi_byte.is_ascii_digit() || !lo_byte.is_ascii_digit() {
            return Err(io::Error::new(
                io::ErrorKind::InvalidData,
                format!("line {line_no}: non-digit byte at pair {}", i + 1),
            ));
        }
        let hi = hi_byte - b'0';
        let lo = lo_byte - b'0';
        let digit = hi * 10 + lo;
        if digit >= 60 {
            return Err(io::Error::new(
                io::ErrorKind::InvalidData,
                format!(
                    "line {line_no}: invalid base-60 digit {digit} at pair {}",
                    i + 1
                ),
            ));
        }
        value = value * 60 + u128::from(digit);
    }
    u64::try_from(value).map_err(|_| {
        io::Error::new(
            io::ErrorKind::InvalidData,
            format!("line {line_no}: decoded value exceeds u64::MAX"),
        )
    })
}
```

The third string `"line {line_no}: non-digit byte at pair {}"` is NEW and unpinned (acceptable — the path is defensive; `find_digit_run` still filters upstream).
