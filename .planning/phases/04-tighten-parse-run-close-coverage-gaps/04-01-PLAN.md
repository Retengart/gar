```rust
pub(crate) fn decode_stream<R: BufRead, W: Write>(
    mut r: R,
    w: &mut W,
    input_format: InputFormat,
) -> io::Result<()> {
    // Buffer the first non-empty line so we can both sniff and replay it.
    let mut first = String::new();
    let mut tried = 0_usize;
    while first.trim().is_empty() && tried < 16 {
        first.clear();
        if r.read_line(&mut first)? == 0 { break; }   // EOF
        tried += 1;
        if !first.trim().is_empty() { break; }
    }

    let fmt = match input_format {
        InputFormat::Auto => sniff(first.trim_start()),
        InputFormat::Ansi | InputFormat::Plain => SniffedFormat::AnsiPlain,
        InputFormat::Json => SniffedFormat::Json,
        InputFormat::Html => SniffedFormat::Html,
    };

    // Replay the buffered first line + remainder via Read::chain.
    let chained = io::Read::chain(io::Cursor::new(first.into_bytes()), r);
    let buffered = io::BufReader::new(chained);

    match fmt {
        SniffedFormat::AnsiPlain => decode_from_text(buffered, w),
        SniffedFormat::Json      => decode_from_json(buffered, w),
        SniffedFormat::Html      => decode_from_html(buffered, w),
    }
}

fn sniff(first_line: &str) -> SniffedFormat {
    let t = first_line.trim_start();
    // Case-insensitive DOCTYPE: HTML_PROLOGUE uses lowercase (format.rs:131),
    // but humans type both.
    if t.starts_with("<!DOCTYPE") || t.starts_with("<!doctype") || t.starts_with("<html") {
        SniffedFormat::Html
    } else if t.starts_with(r#"{"offset":"#) {
        SniffedFormat::Json
    } else {
        SniffedFormat::AnsiPlain
    }
}
```

```rust
fn decode_from_text<R: BufRead, W: Write>(r: R, w: &mut W) -> io::Result<()> {
    let mut trailer: Option<usize> = None;
    let mut buffered_last: Option<[u8; 8]> = None;
    let mut written: usize = 0;

    for (idx, line) in r.lines().enumerate() {
        let line = line?;
        // Trailer check first ŌĆö `# bytes=0x<hex>` with `#` prefix.
        if let Some(hex) = parse_trailer_hex(&line) {
            trailer = Some(hex);
            continue;
        }
        let Some(run) = find_digit_run(&line) else { continue; };
        let run_arr: &[u8; RUN_LEN] = run.as_bytes().try_into()
            .expect("find_digit_run guarantees RUN_LEN bytes");
        let value = parse_run(run_arr, idx + 1)?;
        let bytes = value.to_be_bytes();

        if let Some(prev) = buffered_last.take() {
            w.write_all(&prev)?;
            written += 8;
        }
        buffered_last = Some(bytes);
    }

    // Flush the final buffered chunk, honouring the trailer if present.
    if let Some(last) = buffered_last {
        match trailer {
            Some(total) if total > written => {
                let tail = total.saturating_sub(written).min(8);
                w.write_all(&last[..tail])?;
                written += tail;
            }
            Some(_) => { /* already at or past the trailer length */ }
            None => {
                w.write_all(&last)?;
                written += 8;
            }
        }
    }

    if trailer.is_none() && written > 0 {
        eprintln!(
            "decode: no length metadata; assuming input was 8-byte-aligned. \
             Last chunk may contain zero-padding. Regenerate the dump with \
             base60 v2+ to silence this warning.",
        );
    }

    w.flush()
}

fn parse_trailer_hex(line: &str) -> Option<usize> {
    // Accept `# bytes=0x<hex>` (allowing leading whitespace + trailing garbage).
    let rest = line.trim_start().strip_prefix('#')?.trim_start();
    let rest = rest.strip_prefix("bytes=0x")?;
    let end = rest.find(|c: char| !c.is_ascii_hexdigit()).unwrap_or(rest.len());
    usize::from_str_radix(&rest[..end], 16).ok()
}
```

```rust
fn decode_from_json<R: BufRead, W: Write>(r: R, w: &mut W) -> io::Result<()> {
    let mut expected: Option<usize> = None;
    let mut written: usize = 0;
    for line in r.lines() {
        let line = line?;
        let trimmed = line.trim();
        if let Some(rest) = trimmed.strip_prefix(r#"{"type":"meta","bytes":"#) {
            let end = rest.find(|c: char| !c.is_ascii_digit()).unwrap_or(rest.len());
            expected = rest[..end].parse().ok();
            continue;
        }
        if !trimmed.starts_with(r#"{"offset":"#) { continue; }
        // Find the `"bytes":[` substring and read until the matching `]`.
        let Some(start) = trimmed.find(r#""bytes":["#) else { continue; };
        let after_bracket = start + r#""bytes":["#.len();
        let Some(close) = trimmed[after_bracket..].find(']') else { continue; };
        let inner = &trimmed[after_bracket..after_bracket + close];
        if inner.is_empty() { continue; }
        for token in inner.split(',') {
            let byte: u8 = token.trim().parse().map_err(|_| io::Error::new(
                io::ErrorKind::InvalidData,
                format!("json decode: invalid byte literal {token:?}"),
            ))?;
            w.write_all(&[byte])?;
            written += 1;
        }
    }
    if let Some(total) = expected {
        if total != written {
            eprintln!("decode: json meta bytes={total} but wrote {written}; continuing");
        }
    }
    w.flush()
}
```

```rust
fn decode_from_html<R: BufRead, W: Write>(r: R, w: &mut W) -> io::Result<()> {
    let mut raw = String::new();
    // HTML output is typically small (<10 MiB of dump); read-to-string is
    // acceptable. Streaming HTML parse is Phase 6 territory.
    io::BufReader::new(r).read_to_string(&mut raw)?;

    // Strip outside body.
    let body = raw.find("<body>").map(|i| i + "<body>".len()).unwrap_or(0);
    let end = raw.find("</body>").unwrap_or(raw.len());
    let slice = &raw[body..end];

    let mut trailer: Option<usize> = None;
    // Parse <!-- bytes=0x<hex> --> if present.
    if let Some(idx) = slice.find("<!-- bytes=0x") {
        let rest = &slice[idx + "<!-- bytes=0x".len()..];
        let e = rest.find(|c: char| !c.is_ascii_hexdigit()).unwrap_or(rest.len());
        trailer = usize::from_str_radix(&rest[..e], 16).ok();
    }

    // Collect digit pairs from span tags. We walk char-by-char looking for
    // the three exact span-open prefixes we care about.
    let mut pairs: Vec<u8> = Vec::new();
    let mut cursor = 0;
    while let Some(rel) = slice[cursor..].find("<span class=\"") {
        let abs = cursor + rel + "<span class=\"".len();
        let tail = &slice[abs..];
        let Some(quote_end) = tail.find('"') else { break; };
        let class = &tail[..quote_end];
        let after_open = abs + quote_end + "\">".len();
        if after_open > slice.len() { break; }
        let after = &slice[after_open..];
        let Some(close) = after.find("</span>") else { break; };
        let content = &after[..close];

        if matches!(class, "d-zero" | "d-low" | "d-mid" | "d-high") && content.len() == 2 {
            let bytes = content.as_bytes();
            if bytes[0].is_ascii_digit() && bytes[1].is_ascii_digit() {
                let hi = bytes[0] - b'0';
                let lo = bytes[1] - b'0';
                pairs.push(hi * 10 + lo);
            }
        }
        cursor = after_open + close + "</span>".len();
    }

    // Every 11 pairs = one 8-byte chunk. Drop any trailing partial row.
    let mut written: usize = 0;
    let mut buffered_last: Option<[u8; 8]> = None;
    for row in pairs.chunks_exact(DIGITS) {
        let mut run = [0_u8; RUN_LEN];
        for (i, &d) in row.iter().enumerate() {
            run[i * (PAIR + 1)] = b'0' + d / 10;
            run[i * (PAIR + 1) + 1] = b'0' + d % 10;
            if i + 1 < DIGITS { run[i * (PAIR + 1) + 2] = b':'; }
        }
        let value = parse_run(&run, 0)?;
        let bytes = value.to_be_bytes();
        if let Some(prev) = buffered_last.take() {
            w.write_all(&prev)?;
            written += 8;
        }
        buffered_last = Some(bytes);
    }
    if let Some(last) = buffered_last {
        match trailer {
            Some(total) if total > written => {
                let tail = total.saturating_sub(written).min(8);
                w.write_all(&last[..tail])?;
                written += tail;
            }
            Some(_) => {}
            None => {
                w.write_all(&last)?;
                written += 8;
            }
        }
    }
    let _ = written;
    w.flush()
}
```
